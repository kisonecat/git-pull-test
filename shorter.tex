    \begin{problem}
      And then $A/2 = \answer{1/2}$.
      
      \begin{problem}
        And then $A/4 = \answer{1/4}$.
      \end{problem}
    \end{problem}
  \end{problem}
  
  \begin{problem}

  \end{problem}
